from bs4 import BeautifulSoup def remove_div(html_content): """ Remove specific sections from HTML content using BeautifulSoup """ soup = BeautifulSoup(html_content, 'html.parser') # Remove all the containers we want to remove containers_to_remove = [ ('div', 'table-container'), ('div', 'note-section'), ('div', 'multiplot-container'), ('div', 'plot-container'), ('div', 'constants-container'), ('div', 'download_section') ] for tag, class_name in containers_to_remove: element = soup.find(tag, class_=class_name) if element: element.decompose() # Remove scripts.js script = soup.find('script', {'src': 'scripts.js'}) if script: script.decompose() return str(soup)
