def inject_solver_opts(self, plecs_file_path, output_file_path, solver_opts):
    """
    Replaces solver parameters in the SimulationParameters section of a PLECS file.

    Parameters:
        plecs_file_path     : Path to the source .plecs file
        output_file_path    : Path to the output .plecs file
        solver_opts         : Dict of solver parameters to inject (SolverOpts)
    """

    # Read the entire content of the PLECS file
    with open(plecs_file_path, 'r') as f:
        content = f.read()

    # Map from your SolverOpts keys to the PLECS field names (adjust if needed)
    key_map = {
        'Solver'     : 'Solver',
        'StartTime'  : 'StartTime',
        'TimeSpan'   : 'TimeSpan',
        'Timeout'    : 'Timeout',
        'MaxStep'    : 'MaxStep',
        'FixedStep'  : 'FixedStep',
        'RelTol'     : 'RelTol',
        'Refine'     : 'Refine',
    }

    for key, plecs_key in key_map.items():
        if key not in solver_opts:
            continue

        value = solver_opts[key]

        # Format the value: strings get quotes, numbers stay as-is
        if isinstance(value, str):
            formatted = f'"{value}"'
        else:
            formatted = str(value)

        # Replace existing parameter line: ParameterName  "oldValue" or ParameterName  oldValue
        pattern     = rf'({plecs_key}\s+)(".*?"|\S+)'
        replacement = rf'\g<1>{formatted}'

        if dp.re.search(pattern, content):
            content = dp.re.sub(pattern, replacement, content)
        else:
            # Parameter not found — inject it inside SimulationParameters { ... }
            sim_block_pattern = r'(SimulationParameters\s*\{)'
            content = dp.re.sub(
                sim_block_pattern,
                rf'\1\n    {plecs_key}          {formatted}',
                content
            )

    # Write the modified content to the output file
    with open(output_file_path, 'w') as f:
        f.write(content)
