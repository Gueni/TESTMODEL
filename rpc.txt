def flatten_dict(d, parent='', sep='/'):
    out = {}
    stack = [(parent, d)]

    out_set = out.__setitem__
    stack_append = stack.append

    while stack:
        prefix, current = stack.pop()

        for k, v in current.items():
            key = prefix + sep + k if prefix else k

            if isinstance(v, dict):
                stack_append((key, v))
            else:
                out_set(key, v)

    return out
